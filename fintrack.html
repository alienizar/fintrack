// FinTrack — Apps Script Backend // Auto-sync: mendukung GET (pull) dan POST (push) const SHEET_NAME = 'FinTrack'; function getOrCreateSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(['id','date','desc','amount','type','cat']); sheet.setFrozenRows(1); } return sheet; } function doGet(e) { try { const sheet = getOrCreateSheet(); const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const transactions = rows.slice(1) .filter(r => r[0] && r[3]) .map(r => ({ id: r[0], date: r[1], desc: r[2], amount: Number(r[3]), type: r[4], cat: r[5] })); // Baca metadata (budgets, cats) dari sheet tersendiri const ss = SpreadsheetApp.getActiveSpreadsheet(); let metaSheet = ss.getSheetByName('FinTrack_Meta'); let budgets = [], cats = []; if (metaSheet) { const metaRows = metaSheet.getDataRange().getValues(); metaRows.forEach(r => { try { if (r[0] === 'budgets') budgets = JSON.parse(r[1]); if (r[0] === 'cats') cats = JSON.parse(r[1]); } catch(e) {} }); } const result = JSON.stringify({ ok: true, transactions, budgets, cats }); return ContentService.createTextOutput(result) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ ok: false, error: err.message })) .setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { try { const payload = JSON.parse(e.postData.contents); const sheet = getOrCreateSheet(); const ss = SpreadsheetApp.getActiveSpreadsheet(); // Tulis transactions sheet.clearContents(); sheet.appendRow(['id','date','desc','amount','type','cat']); (payload.transactions || []).forEach(t => { sheet.appendRow([t.id, t.date, t.desc, t.amount, t.type, t.cat]); }); // Tulis metadata ke sheet terpisah let metaSheet = ss.getSheetByName('FinTrack_Meta'); if (!metaSheet) metaSheet = ss.insertSheet('FinTrack_Meta'); metaSheet.clearContents(); metaSheet.appendRow(['budgets', JSON.stringify(payload.budgets || [])]); metaSheet.appendRow(['cats', JSON.stringify(payload.cats || [])]); return ContentService.createTextOutput(JSON.stringify({ ok: true, rows: sheet.getLastRow() - 1, synced_at: new Date().toISOString() })).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ ok: false, error: err.message })) .setMimeType(ContentService.MimeType.JSON); } }
